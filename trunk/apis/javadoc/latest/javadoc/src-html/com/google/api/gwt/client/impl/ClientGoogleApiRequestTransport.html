016    package com.google.api.gwt.client.impl;
017    
018    import com.google.api.gwt.shared.GoogleApiRequestTransport;
019    import com.google.gwt.core.client.Callback;
020    import com.google.gwt.core.client.JavaScriptObject;
021    import com.google.gwt.core.client.ScriptInjector;
022    import com.google.gwt.http.client.Response;
023    import com.google.web.bindery.requestfactory.shared.Receiver;
024    import com.google.web.bindery.requestfactory.shared.ServerFailure;
025    
026    import java.util.Map;
027    
028    /**
029     * Implementation of {@link GoogleApiRequestTransport} that should be used in
030     * the browser. It defers requests to a the JS library which provides
031     * cross-domain RPC.
032     */
033    public class ClientGoogleApiRequestTransport extends GoogleApiRequestTransport {
034    
035      private static final String JS_CLIENT_NAME = "ae_f85e3bd0744c7080861c3ae42085d071.js";
036      private static final String JS_CLIENT_URL = "https://ssl.gstatic.com/gb/js/" + JS_CLIENT_NAME;
037    
038      private static final class ApiResponseJso extends JavaScriptObject {
039        @SuppressWarnings("unused")
040        protected ApiResponseJso() {
041        }
042    
043        final native String getBody() /*-{
044          return this.body;
045        }-*/;
046    
047        final native int getStatus() /*-{
048          return this.status;
049        }-*/;
050    
051        final native String getStatusText() /*-{
052          return this.statusText;
053        }-*/;
054      }
055    
056      /** Called from JSNI callback in {@link #makeCall}. */
057      protected static void handleResult(JavaScriptObject result, TransportReceiver receiver) {
058        ApiResponseJso response = result.cast();
059        if (response.getStatus() != Response.SC_OK) {
060          receiver.onTransportFailure(
061              new ServerFailure(response.getStatus() + " " + response.getStatusText()));
062        } else {
063          receiver.onTransportSuccess(response.getBody());
064        }
065      }
066    
067      protected static native boolean isLoaded() /*-{
068        return !!$wnd.googleapis && !!$wnd.googleapis.newHttpRequest;
069      }-*/;
070    
071      private static native void makeCall(JavaScriptObject data, TransportReceiver receiver) /*-{
072        var callback = $entry(
073        function(result) {
074          @com.google.api.gwt.client.impl.ClientGoogleApiRequestTransport::handleResult(*)
075            (result, receiver);
076        });
077        $wnd.googleapis.newHttpRequest(data).execute(callback);
078      }-*/;
079    
080      private static native JavaScriptObject makeData(
081          String url, JavaScriptObject headers, String payload) /*-{
082        return {
083            'url' : url,
084            'headers' : headers,
085            'body' : payload,
086            'httpMethod' : 'POST'
087        };
088      }-*/;
089    
090      private static native void set(JavaScriptObject jso, String key, String value) /*-{
091        jso[key] = value;
092      }-*/;
093    
094      protected final JavaScriptObject headers = JavaScriptObject.createObject();
095    
096      @Override
097      protected void configure(final Receiver<GoogleApiRequestTransport> receiver) {
098        // Only load the script once, if the necessary API isn't available
099        if (!isLoaded()) {
100          ScriptInjector.fromUrl(JS_CLIENT_URL).setCallback(new Callback<Void, Exception>() {
101            public void onSuccess(Void result) {
102              baseConfigure(baseUrl);
103              setHeaders();
104              receiver.onSuccess(ClientGoogleApiRequestTransport.this);
105            }
106    
107            public void onFailure(Exception e) {
108              receiver.onFailure(new ServerFailure(e.getMessage()));
109            }
110          }).inject();
111        } else {
112          receiver.onSuccess(this);
113        }
114      }
115    
116      private void setHeaders() {
117        for (Map.Entry<String, String> entry : getHeaders().entrySet()) {
118          set(headers, entry.getKey(), entry.getValue());
119        }
120    
121      }
122    
123      /** Configures the base JS client to use the alternate JS library. */
124      private static native void baseConfigure(String baseUrl) /*-{
125        $wnd['__GOOGLEAPIS'] = {
126          'googleapis.config' : {
127            'proxy': baseUrl + "/static/proxy.html",
128            'gcv' : @com.google.api.gwt.client.impl.ClientGoogleApiRequestTransport::JS_CLIENT_NAME
129          }
130        };
131      }-*/;
132    
133      @Override
134      public void send(String payload, TransportReceiver receiver) {
135        JavaScriptObject data = makeData(getRpcUrl(), headers, payload);
136        makeCall(data, receiver);
137      }
138    }
