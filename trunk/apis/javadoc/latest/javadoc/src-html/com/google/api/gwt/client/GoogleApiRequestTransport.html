015    package com.google.api.gwt.client;
016    
017    import com.google.gwt.core.client.Callback;
018    import com.google.gwt.core.client.JavaScriptObject;
019    import com.google.gwt.core.client.JsonUtils;
020    import com.google.gwt.http.client.Response;
021    import com.google.web.bindery.requestfactory.shared.RequestTransport;
022    import com.google.web.bindery.requestfactory.shared.ServerFailure;
023    
024    /**
025     * {@link RequestTransport} designed to communicate with Google APIs.
026     *
027     * @author jasonhall@google.com (Jason Hall)
028     */
029    public class GoogleApiRequestTransport implements RequestTransport {
030    
031      private final String apiKey;
032      private final String userAgent;
033    
034      public GoogleApiRequestTransport(String applicationName, String apiKey) {
035        this(applicationName, apiKey, "https://www.googleapis.com");
036      }
037    
038      public GoogleApiRequestTransport(String applicationName, String apiKey, String baseUrl) {
039        this.apiKey = apiKey;
040        this.userAgent =
041            (applicationName == null ? "" : (applicationName + " ")) + "google-api-gwt-client/0.1";
042    
043        baseConfigure(baseUrl);
044      }
045    
046      private static native void baseConfigure(String baseUrl) /*-{
047        $wnd['__GOOGLEAPIS'] = {
048          'googleapis.config' : {
049            'proxy': baseUrl + "/static/proxy.html",
050          }
051        };
052      }-*/;
053    
054      private native void nativeSend(String payload, TransportReceiver receiver) /*-{
055        var callback =  $entry(function(result) {
056          @com.google.api.gwt.client.GoogleApiRequestTransport::handleResult(*)
057          (result, receiver);
058        });
059    
060        // TODO(jasonhall): Find some way to avoid this.
061        eval('var payloadObject = ' + payload);
062    
063        var requestObj = {
064          "url": "/rpc",
065          "headers": {
066            "Content-Type": "application/json",
067            "X-JavaScript-User-Agent":
068                this.@com.google.api.gwt.client.GoogleApiRequestTransport::userAgent
069          },
070          "httpMethod": "POST",
071          "body": {
072            "method": payloadObject.method,
073            "apiVersion": payloadObject.apiVersion,
074            "params": payloadObject.params
075          }
076        }
077        requestObj['body']['params']['key'] =
078            this.@com.google.api.gwt.client.GoogleApiRequestTransport::apiKey;
079    
080        $wnd.googleapis.newHttpRequest(requestObj).execute(callback);
081      }-*/;
082    
083      private static final class ApiResponseJso extends JavaScriptObject {
084        @SuppressWarnings("unused")
085        protected ApiResponseJso() {
086        }
087    
088        final native String getBody() /*-{
089          return this.body;
090        }-*/;
091    
092        final native int getStatus() /*-{
093          return this.status;
094        }-*/;
095    
096        final native String getStatusText() /*-{
097          return this.statusText;
098        }-*/;
099      }
100    
101      private static void handleResult(JavaScriptObject result, TransportReceiver receiver) {
102        ApiResponseJso response = result.cast();
103    
104        if (response.getStatus() == Response.SC_OK) {
105          receiver.onTransportSuccess(response.getBody());
106        } else {
107          receiver.onTransportFailure(
108              new ServerFailure(response.getStatus() + " " + response.getStatusText()));
109        }
110      }
111    
112      @Override
113      public void send(final String payload, final TransportReceiver receiver) {
114        // Ensure that the API library is loaded. If it is, this callback will be executed immediately.
115        GoogleApiLoader.get().load(new Callback<Void, Exception>() {
116          @Override
117          public void onSuccess(Void v) {
118            makeRequest(payload, receiver);
119          }
120    
121          @Override
122          public void onFailure(Exception e) {
123            receiver.onTransportFailure(new ServerFailure(e.getMessage()));
124          }
125        });
126      }
127    
128      private void makeRequest(String payload, TransportReceiver receiver) {
129        if (JsonUtils.safeToEval(payload)) {
130          nativeSend(payload, receiver);
131        } else {
132          receiver.onTransportFailure(new ServerFailure("Request payload is invalid."));
133        }
134      }
135    }
