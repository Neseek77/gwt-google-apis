016    package com.google.api.gwt.shared;
017    
018    import com.google.web.bindery.requestfactory.shared.Receiver;
019    import com.google.web.bindery.requestfactory.shared.RequestTransport;
020    import com.google.web.bindery.requestfactory.shared.ServerFailure;
021    
022    import java.util.Collections;
023    import java.util.HashMap;
024    import java.util.Map;
025    
026    /**
027     * Provides access to Google API endpoints for RequestFactory.
028     *
029     */
030    public abstract class GoogleApiRequestTransport implements RequestTransport {
031    
032      private String applicationName;
033      private Map<String, String> queryParams = new HashMap<String, String>();
034      private Map<String, String> headers = new HashMap<String, String>();
035      protected String baseUrl = "https://www.googleapis.com";
036      private String rpcUrl = baseUrl + "/rpc";
037    
038      private boolean finished = false;
039    
040      /**
041       * Initialize the RequestTransport and pass it to the {@code receiver}
042       * parameter.
043       */
044      public void create(Receiver<GoogleApiRequestTransport> receiver) {
045        if (finished) {
046          receiver.onFailure(new ServerFailure("Already called create() on GoogleApiRequestTransport"));
047          return;
048        }
049        StringBuilder currentUrl = new StringBuilder(baseUrl + "/rpc");
050        boolean needsAmp = currentUrl.indexOf("?") != -1;
051        for (Map.Entry<String, String> entry : queryParams.entrySet()) {
052          if (needsAmp) {
053            currentUrl.append("&");
054          } else {
055            currentUrl.append("?");
056            needsAmp = true;
057          }
058          currentUrl.append(entry.getKey()).append("=").append(entry.getValue());
059        }
060        rpcUrl = currentUrl.toString();
061    
062        // Mandatory headers
063        String userAgent = (applicationName == null ? "" : (applicationName + " ")) + USER_AGENT_STRING;
064        setRequestHeader(CONTENT_TYPE_HEADER, CONTENT_TYPE);
065        setRequestHeader(USER_AGENT_HEADER, userAgent);
066        headers = Collections.unmodifiableMap(headers);
067        queryParams = Collections.unmodifiableMap(new HashMap<String, String>());
068    
069        configure(receiver);
070      }
071    
072      /**
073       * Set the OAuth access token to pass with the request in order to access
074       * protected information.
075       */
076      public GoogleApiRequestTransport setAccessToken(String authorization) {
077        setRequestHeader(AUTHORIZATION_HEADER, OAUTH + authorization);
078        return this;
079      }
080    
081      /**
082       * Set the API access key to support quota management.
083       */
084      public GoogleApiRequestTransport setApiAccessKey(String apiAccessKey) {
085        setQueryParameter("key", apiAccessKey);
086        return this;
087      }
088    
089      /**
090       * Set the application name for use in quota management.
091       *
092       * @param applicationName A user-agent fragment, such as {@code MyApp/1.0}
093       */
094      public GoogleApiRequestTransport setApplicationName(String applicationName) {
095        this.applicationName = applicationName;
096        return this;
097      }
098    
099      /**
100       * Append a query parameter to the URL used to make the request.
101       */
102      public GoogleApiRequestTransport setQueryParameter(String name, String value) {
103        if (value == null) {
104          queryParams.remove(name);
105        } else {
106          queryParams.put(name, value);
107        }
108        return this;
109      }
110    
111      /**
112       * Add a header to every HTTP query used to make a request.
113       */
114      public GoogleApiRequestTransport setRequestHeader(String name, String value) {
115        if (value == null) {
116          headers.remove(name);
117        } else {
118          headers.put(name, value);
119        }
120        return this;
121      }
122    
123      private static final String AUTHORIZATION_HEADER = "Authorization";
124      private static final String CONTENT_TYPE = "application/json";
125      private static final String CONTENT_TYPE_HEADER = "Content-Type";
126      private static final String OAUTH = "OAuth ";
127      private static final String USER_AGENT_HEADER = "X-JavaScript-User-Agent";
128      private static final String USER_AGENT_STRING = "google-api-gwt-client/0.1";
129    
130      protected GoogleApiRequestTransport() {
131      }
132    
133      protected abstract void configure(Receiver<GoogleApiRequestTransport> receiver);
134    
135      protected Map<String, String> getHeaders() {
136        return headers;
137      }
138    
139      protected String getRpcUrl() {
140        return rpcUrl;
141      }
142    
143      /**
144       * Override the RPC URL being requested. By default this is
145       * {@link "https://www.googleapis.com/"}. If you set this, it must *not* include
146       * the trailing slash.
147       */
148      public GoogleApiRequestTransport setBaseUrl(String baseUrl) {
149        this.baseUrl = baseUrl;
150        return this;
151      }
152    }
